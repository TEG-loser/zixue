# 一、最终代码：thread_pool.hpp（核心）
这是**工业级简化版线程池**，结构和 progschj/ThreadPool 完全一致。

```cpp
#ifndef THREAD_POOL_HPP
#define THREAD_POOL_HPP

#include <iostream>
#include <vector>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <functional>
#include <thread>
#include <future>
#include <stdexcept>

class ThreadPool {
public:
    // 构造：启动 thread_num 个工作线程
    ThreadPool(size_t thread_num);

    // 析构：优雅退出
    ~ThreadPool();

    // 提交任务（核心接口）
    template<class F, class... Args>
    auto enqueue(F&& f, Args&&... args) 
        -> std::future<typename std::result_of<F(Args...)>::type>;

    // 禁用拷贝
    ThreadPool(const ThreadPool&) = delete;
    ThreadPool& operator=(const ThreadPool&) = delete;

private:
    // 工作线程列表
    std::vector<std::thread> workers_;
    
    // 任务队列
    std::queue<std::function<void()>> tasks_;
    
    // 同步
    std::mutex mutex_;
    std::condition_variable cv_;
    
    // 退出标志
    bool stop_;
};

// ===================== 实现 =====================
inline ThreadPool::ThreadPool(size_t thread_num) : stop_(false) {
    for (size_t i = 0; i < thread_num; ++i) {
        workers_.emplace_back([this] {
            while (true) {
                std::function<void()> task;

                {
                    std::unique_lock<std::mutex> lock(this->mutex_);
                    // 等待：停止 或 有任务
                    this->cv_.wait(lock, [this] {
                        return this->stop_ || !this->tasks_.empty();
                    });

                    // 优雅退出
                    if (this->stop_ && this->tasks_.empty())
                        return;

                    // 取任务
                    task = std::move(this->tasks_.front());
                    this->tasks_.pop();
                }

                // 执行任务（异常安全）
                try {
                    task();
                } catch (...) {
                    // 捕获异常，不崩溃线程池
                    std::cerr << "[ThreadPool] task exception caught\n";
                }
            }
        });
    }
}

// 提交任务
template<class F, class... Args>
auto ThreadPool::enqueue(F&& f, Args&&... args) 
    -> std::future<typename std::result_of<F(Args...)>::type> {
    
    using ReturnType = typename std::result_of<F(Args...)>::type;

    auto task = std::make_shared<std::packaged_task<ReturnType()>>(
        std::bind(std::forward<F>(f), std::forward<Args>(args)...)
    );

    std::future<ReturnType> res = task->get_future();

    {
        std::unique_lock<std::mutex> lock(mutex_);

        // 禁止向已停止的线程池提交任务
        if (stop_)
            throw std::runtime_error("enqueue on stopped ThreadPool");

        tasks_.emplace([task]() { (*task)(); });
    }

    cv_.notify_one(); // 唤醒一个线程
    return res;
}

// 析构 + 优雅退出
inline ThreadPool::~ThreadPool() {
    {
        std::unique_lock<std::mutex> lock(mutex_);
        stop_ = true;
    }
    cv_.notify_all(); // 唤醒所有线程

    // 等待所有线程结束
    for (std::thread& worker : workers_) {
        if (worker.joinable())
            worker.join();
    }
}

#endif
```

# 二、测试代码：main.cpp（验证功能）
```cpp
#include "thread_pool.hpp"
#include <iostream>
#include <chrono>

using namespace std;

int main() {
    // 创建 4 线程的线程池
    ThreadPool pool(4);

    // 提交 8 个任务
    vector<future<int>> results;

    for (int i = 0; i < 8; ++i) {
        results.emplace_back(
            pool.enqueue([i] {
                cout << "Task " << i << " running..." << endl;
                this_thread::sleep_for(chrono::seconds(1));
                return i * 10;
            })
        );
    }

    // 获取结果
    cout << "\nResults:\n";
    for (auto& res : results) {
        cout << res.get() << " ";
    }
    cout << "\n\nAll tasks done\n";

    return 0;
}
```

# 六、最终总结（可直接写进博客）
# 线程池总结
## 1. 核心思想
- 线程池 = **固定线程 + 任务队列 + 条件变量**
- 避免频繁创建/销毁线程
- 实现任务异步执行

## 2. 同步机制
- 互斥锁保护任务队列
- 条件变量实现**等待/唤醒**
- 无空轮询，CPU 利用率高

## 3. 关键设计
- `wait(lock, predicate)` 必须用条件判断
- 任务执行加 **try-catch** 保证异常安全
- 析构函数实现**优雅退出**
- 所有线程 join 后再退出

## 4. 适用场景
- 高并发短任务
- 网络服务器
- 异步任务处理
