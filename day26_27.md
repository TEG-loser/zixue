# 最终完整代码：myshell.cpp
```cpp
#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <unistd.h>
#include <sys/wait.h>
#include <signal.h>
#include <vector>
#include <sstream>

using namespace std;

// 分割命令字符串（按空格）
vector<string> split(const string &s) {
    vector<string> res;
    stringstream ss(s);
    string word;
    while (ss >> word) {
        res.push_back(word);
    }
    return res;
}

// 执行外部命令
void run_external(vector<string> &args) {
    pid_t pid = fork();

    if (pid == 0) {
        // 子进程 exec
        int argc = args.size();
        char **argv = new char*[argc + 1];
        for (int i = 0; i < argc; ++i) {
            argv[i] = &args[i][0];
        }
        argv[argc] = nullptr;

        execvp(argv[0], argv);
        perror("exec failed");
        exit(1);
    } else if (pid > 0) {
        wait(NULL);
    } else {
        perror("fork failed");
    }
}

// 内置命令：cd
void builtin_cd(vector<string> &args) {
    if (args.size() < 2) {
        chdir(getenv("HOME"));
    } else {
        if (chdir(args[1].c_str()) == -1) {
            perror("cd error");
        }
    }
}

// 执行命令（内置 + 外部）
void execute(string line) {
    auto args = split(line);
    if (args.empty()) return;

    if (args[0] == "cd") {
        builtin_cd(args);
    } else if (args[0] == "exit") {
        exit(0);
    } else {
        run_external(args);
    }
}

int main() {
    // 忽略 Ctrl+C 让 shell 不退出
    signal(SIGINT, SIG_IGN);

    string line;
    while (true) {
        printf("[myshell] $ ");
        fflush(stdout);

        if (!getline(cin, line)) break;
        execute(line);
    }
    return 0;
}
```

# Shell 核心原理总结（面试必问）
## 1. Shell 做什么？
1. 打印提示符 `$`
2. 读取命令
3. 解析命令
4. **内置命令（cd/exit）直接执行**
5. **外部命令 fork + exec 执行**
6. 父进程 wait 等待子进程结束

## 2. 为什么 cd 必须是内置命令？
- `cd` 要改变**当前进程所在目录**
- 如果用 fork + exec，子进程改目录，父进程不变 → 无效
- 所以必须**内置在 Shell 进程内执行**

## 3. 简易 Shell 执行流程（伪代码）
```
while (true) {
    打印提示符
    读取一行命令
    分割命令为参数列表

    if 是 cd/exit：
        内置执行
    else
        fork()
        子进程：execvp()
        父进程：wait()
}
```
